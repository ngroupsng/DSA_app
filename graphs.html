Graph Traversal — BFS • DFS

A Graph is a collection of nodes (vertices) connected by edges. Traversal means visiting every vertex once in a specific order.

  • BFS (Breadth-First Search): Level by level using a queue.
  • DFS (Depth-First Search): Goes deep using recursion or a stack.
BFS(G, start):
  create queue Q
  mark start visited
  Q.enqueue(start)
  while Q not empty:
    v = Q.dequeue()
    for each neighbor u of v:
      if not visited[u]:
        visited[u] = true
        Q.enqueue(u)

DFS(G, v):
  mark v visited
  for each neighbor u of v:
    if not visited[u]:
      DFS(G, u)

BFS Program in C


#include <stdio.h>
#include <stdlib.h>

#define MAX 20

int queue[MAX], front = 0, rear = -1;

void bfs(int a[MAX][MAX], int visited[MAX], int start, int n) {
    int i;
    visited[start] = 1;
    queue[++rear] = start;

    while (front <= rear) {
        int u = queue[front++];
        printf("%d ", u);

        for (i = 0; i < n; i++) {
            if (a[u][i] == 1 && visited[i] == 0) {
                visited[i] = 1;
                queue[++rear] = i;
            }
        }
    }
}

int main() {
    int a[MAX][MAX], visited[MAX] = {0};
    int n, i, j, start;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");
    for (i = 0; i < n; i++)
        for (j = 0; j < n; j++)
            scanf("%d", &a[i][j]);

    printf("Enter starting vertex: ");
    scanf("%d", &start);

    printf("BFS Traversal: ");
    bfs(a, visited, start, n);
    return 0;
}

#include <stdio.h>

#define MAX 20

void dfs(int a[MAX][MAX], int visited[MAX], int start, int n) {
    int i;
    printf("%d ", start);
    visited[start] = 1;

    for (i = 0; i < n; i++) {
        if (a[start][i] == 1 && visited[i] == 0) {
            dfs(a, visited, i, n);
        }
    }
}
