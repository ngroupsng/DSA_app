Binary Tree — Insertion & Deletion

A Binary Tree is a hierarchical structure where each node has at most two children: left and right. Insertions happen level-wise here.

INSERT(root, x):
  new = node(x)
  if root == NULL:
    root = new
  else:
    q = queue()
    q.push(root)
    while q not empty:
      temp = q.pop()
      if temp.left == NULL:
        temp.left = new; break
      else q.push(temp.left)
      if temp.right == NULL:
        temp.right = new; break
      else q.push(temp.right)

struct Node { int data; struct Node *left,*right; };
struct Node* root = NULL;
void insert(int val){
  struct Node* new = malloc(sizeof(struct Node));
  new->data=val; new->left=new->right=NULL;
  if(root==NULL){root=new;return;}
  queue q; push(root);
  while(!empty(q)){
    temp=pop(q);
    if(temp->left==NULL){temp->left=new;break;}
    else push(temp->left);
    if(temp->right==NULL){temp->right=new;break;}
    else push(temp->right);
  }
}
